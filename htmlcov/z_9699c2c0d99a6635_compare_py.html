18_ENCODING = tiktoken.get_encoding("cl100k_base") # duplicates represent._ENCODING; features may not import each other

22class CompareResult(BaseModel): 

23 """Fidelity of an edit: how close ``edited`` stays to ``original`` and how much shorter it got.""" 

24 

25 model_config = ConfigDict(frozen=True) 

26 similarity: float # cosine similarity of the two chunk-pooled embeddings 

27 original_tokens: int # cl100k_base 

28 edited_tokens: int 

29 token_reduction: float # 1 - edited_tokens / original_tokens (negative when edited is longer)

31 @computed_field 

32 @property 

33 def cos_sim_diff(self) -> CosSimDiff: 

34 # pydantic does not validate a computed_field's return, so this clamp is the real ge=0 guard. 

35 return max(0.0, 1.0 - self.similarity) 

36

38def compare(original: str, edited: str, embedder: Embedder | None = None) -> CompareResult: 

39 """Compare two prompts: chunk-pooled cosine similarity and cl100k_base token reduction. 

40 

41 The caller applies the configured gate by comparing ``similarity`` against ``1 - Params.cos_sim_diff_budget``. 

42 When ``embedder`` is omitted, the default OpenAI embedder is built on first use (requires an API key). 

43 """ 

44 model = embedder if embedder is not None else default_embedder() 

45 original_ids = _ENCODING.encode(original, disallowed_special=()) 

46 edited_ids = _ENCODING.encode(edited, disallowed_special=()) 

47 if not original_ids or not edited_ids: 

48 raise ValueError("cannot compare an empty text") 

49 similarity = 1.0 - compute_cos_sim_diff( 

50 _pooled_embedding(original_ids, model), _pooled_embedding(edited_ids, model) 

51 ) 

52 return CompareResult( 

53 similarity=similarity, 

54 original_tokens=len(original_ids), 

55 edited_tokens=len(edited_ids), 

56 token_reduction=1.0 - len(edited_ids) / len(original_ids), 

57 ) 

58

60def _pooled_embedding(token_ids: list[int], embedder: Embedder) -> NDArray[np.float32]: 

61 """The length-weighted mean of the normalized ~200-token chunk vectors. 

62 

63 Chunk pooling keeps long prompts comparable across embedding backends with modest context windows; 

64 the 200-token chunk size is retained from the previous local-model backend. Embedding a long text 

65 as one vector would blind the gate, so chunking keeps every token in view. Any Embedder is chunked — 

66 the protocol exposes no window. 

67 """ 

68 chunks = [token_ids[start : start + _CHUNK_TOKENS] for start in range(0, len(token_ids), _CHUNK_TOKENS)] 

69 weights = np.array([len(chunk) for chunk in chunks], dtype=np.float32) 

70 vectors = normalize(np.stack(embedder.embed([_ENCODING.decode(chunk) for chunk in chunks]))) 

71 return ((vectors * weights[:, None]).sum(axis=0) / weights.sum()).astype(np.float32)
