320def _location(diff: Diff) -> str: 

321 path = diff.spans[0].section_path 

322 return " > ".join(part for part in path if part) or "(top level)" 

323 

324 

325def _hunk_html(document: Document, removed: frozenset[SentenceId]) -> str: 

326 parts: list[str] = [] 

327 for kind, text in hunk_lines(document, removed): 

328 escaped = html.escape(text) 

329 if kind == "removed": 

330 parts.append(f'<div class="line removed">-{escaped}</div>') 

331 elif kind == "context": 331 ↛ 333line 331 didn't jump to line 333 because the condition on line 331 was always true

332 parts.append(f'<div class="line context"> {escaped}</div>') 

333 elif kind == "gap": 

334 parts.append(f'<div class="line gap">{escaped}</div>') 

335 else: 

336 parts.append(f'<div class="line empty-msg">{escaped}</div>') 

337 return "\n".join(parts)

340def _edit_block_html(document: Document, index: int, diff: Diff) -> str: 

341 candidate = diff.candidate 

342 targets = frozenset(candidate.edit.targets) 

343 hunk = _hunk_html(document, targets) 

344 if diff.replacement: 

345 # Appended after the whole hunk (not interleaved via hunk_lines' `added`) to keep the 

346 # replacement line at the block's end, where this per-edit view has always shown it. 

347 hunk += f'\n<div class="line added">+{html.escape(diff.replacement)}</div>' 

348 confidence = f"{candidate.confidence:.3f}" 

349 return f"""<article class="edit-block rejected" data-index="{index}"> 

350 <div class="edit-meta"> 

351 <div class="loc">{html.escape(_location(diff))}</div> 

352 <div class="detail">confidence: {confidence} · optimizer: {html.escape(candidate.source)}<br> 

353 reason: {html.escape(candidate.reason)}</div> 

354 </div> 

355 <div class="hunk">{hunk}</div> 

356 <div class="controls"> 

357 <button type="button" class="accept">Accept</button> 

358 <button type="button" class="reject active">Reject</button> 

359 </div> 

360</article>"""

363def _diff_preview(diff: Diff) -> DiffPreview: 

364 edit = diff.candidate.edit 

365 replacement_token_count = edit.replacement.token_count if edit.op == "replace" else 0 

366 return DiffPreview(targets=edit.targets, replacement_token_count=replacement_token_count)

369def _document_preview(document: Document) -> DocumentPreview: 

370 return DocumentPreview( 

371 token_count=document.token_count, 

372 sentences=tuple( 

373 SentencePreview(id=sentence.id, text=sentence.text, token_count=sentence.token_count) 

374 for sentence in document.sentences 

375 ), 

376 )

379def _initial_selection(total: int) -> dict[str, object]: 

380 return { 

381 "schema_version": SCHEMA_VERSION, 

382 "accepted_indices": [], 

383 "accepted_count": 0, 

384 "total_count": total,

388def render_review_page(proposal: Proposal) -> str: 

389 """Render a self-contained HTML page for reviewing proposed edits.""" 

390 payload = ReviewPayload( 

391 document=_document_preview(proposal.document), 

392 diffs=tuple(_diff_preview(diff) for diff in proposal.diffs), 

393 ) 

394 payload_json = json.dumps(payload.model_dump(mode="json")) 

395 selection_json = json.dumps(_initial_selection(len(proposal.diffs))) 

396 

397 if not proposal.diffs: 

398 edit_blocks = '<p class="empty">no proposed edits; the prompt is unchanged</p>' 

399 shortcuts = "" 

400 else: 

401 edit_blocks = "\n".join( 

402 _edit_block_html(proposal.document, index, diff) for index, diff in enumerate(proposal.diffs) 

403 ) 

404 shortcuts = """<div class="shortcuts"> 

405 <button type="button" id="accept-all">Accept all</button> 

406 <button type="button" id="reject-all">Reject all</button> 

407 </div>""" 

408 

409 return ( 

410 TEMPLATE.replace("__PAYLOAD__", payload_json) 

411 .replace("__SELECTION__", selection_json) 

412 .replace("__EDIT_BLOCKS__", edit_blocks) 

413 .replace("__SHORTCUTS__", shortcuts) 

414 )
